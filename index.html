--------- join method: ---------

1. i/p :var fruits=['Apple','Banana','Orange']
solution: console.log(fruits.join(" ,"));
O/P :"Apple,Banana,Orange"

2. i/p: var numbers=[1,2,3,4,5];
Solution : console.log(numbers.join("-"));
O/P : 1-2-3-4-5

3.var chars=['H','E','L','L','O'];
Solution: console.log(chars.join(""));
O/P : HELLO

---------slice method: --------

4.var text='Javascript is awesome';
Solution: console.log(text.slice(0,10));
O/P: Javascript

5.var sentence= 'I love learning Javascript!';
Solution: console.log(sentence.slice(16,26));
O/P: Javascript

6. var text2='Frontend Development!';(Using negative indexes)
Solution: console.log(text2.slice(-12,-1));
O/P: Development

------split method --------

7.var date='2024-10-21';
Solution: console.log(date.split("-"));
O/P:['2024','10','21']

8. var text2='I love Javascript Programming';
Solution: console.log(text2.split(" ").slice(0,2));
O/P: ['I','love']

9. let url='https://www.example.com/path/page.html';
Solution: console.log(url.split("/"))
O/P:['https:', '', 'www.example.com', 'path', 'page.html']
